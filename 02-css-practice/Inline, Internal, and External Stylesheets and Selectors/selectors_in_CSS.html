HTML and CSS Code Example

        <!DOCTYPE html>
        <html lang="en">
        <head>
            <meta charset="UTF-8">
            <meta name="viewport" content="width=device-width, initial-scale=1.0">
            <title>Selectors in CSS</title>
            <style>
                /* Element Selector */
                h1 {
                    color: blue;
                }
                p{
                    color: blueviolet;
                }
                
                /* Class Selector */
                .highlight {
                    background-color: yellow;
                }
                
                /* ID Selector */
                #important {
                    color: red;
                    font-weight: bold;
                }
                
                /* Descendant Selector */
                .container p {
                    font-style: italic;
                    color: black;
                }
                
                /* Combinator Selectors */
                /* Descendant Selector */
                div p {
                    font-weight: bold;
                }
                /* Child Selector */
                ul > li {
                    color: green;
                }
                /* Adjacent Sibling Selector */
                h2 + p {
                    font-style: italic;
                }
                /* General Sibling Selector */
                h2 ~ p {
                    color: purple;
                }
                
                /* Pseudo-class Selectors */
                /* Hover */
                p:hover {
                    background-color: lightblue;
                }
                /* Active */
                p:active {
                    color: orange;
                }
                /* Focus */
                input:focus {
                    border: 2px solid red;
                }
                /* nth-child */
                ul li:nth-child(odd) {
                    background-color: lightgray;
                }
                /* First Child */
                ul li:first-child {
                    font-weight: bold;
                }
                /* Last Child */
                ul li:last-child {
                    font-style: italic;
                }
                /* Not Selector */
                p:not(.highlight) {
                    color: gray;
                }
                
                /* Pseudo-elements Selectors */
                /* Before */
                p::before {
                    content: "Before Text ";
                }
                /* After */
                p::after {
                    content: " After Text";
                }
                /* First Line */
                p::first-line {
                    font-weight: bold;
                }
                /* First Letter */
                p::first-letter {
                    font-size: 24px;
                }
                
                /* Attribute Selectors */
                /* Attribute */
                a[target] {
                    color: blue;
                }
                /* Attribute with Value */
                input[type="text"] {
                    border: 1px solid black;
                }
                /* Attribute with Word */
                a[href*="example"] {
                    text-decoration: underline;
                }
                /* Attribute Starting with Value */
                input[type^="sub"] {
                    background-color: lightgray;
                }
                /* Attribute Ending with Value */
                input[type$="sword"] {
                    border: 2px dashed green;
                }
                /* Attribute Containing Value */
                input[type*="pass"] {
                    background-color: lightgreen;
                }
                </style>
                </head>
                <body>

                    <h2>ID Selector</h2>
                    <p id="important">This paragraph is styled to be important using an ID selector.</p>
                    
                    <h2>Descendant Selector</h2>
                    <div class="container">
                        <p>This paragraph is italicized because it is inside a container with a specific class.</p>
                    </div>

                        <li>This is a child item</li>
                        <li>This is another child item</li>
                    </ul>
